#version 330

layout (location = 0) in vec3 pos;
layout (location = 1) in vec2 texPos;
layout (location = 4) in vec3 normal;

uniform mat4 projMatrix;
uniform mat4 viewMatrix;

out vec2 otexPos;
out vec3 onormal;

void main() {
	gl_Position = projMatrix * viewMatrix * vec4(pos, 1.0);
	otexPos = texPos;
	onormal = normal;
	
}
